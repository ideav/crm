      "flag": "gb"
    }
  ],
  "teams": [
    {
      "id": "mancity",
      "name": "Манчестер Сити",
      "league_id": "epl",
      "is_real": true,
      "owner": "system",
      "base_tr": 4500
    }
  ],
  "players": [
    {
      "id": "erling-haaland",
